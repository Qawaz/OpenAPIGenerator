Success  {
	success : boolean
    }



   Error  {
	error : string
	message : string
    }



   AppInformation  {
	package_name : string
	version_number : int
	version : string
    }



   UserInformation  {
	user_email : string
	user_id : string
	user_name : string
    }



   DeviceInformation  {
	platform_type : string
	os_version : string
	os_name : string
	info_map : map<string, string>
    }



   DisplayInformation  {
	orientation : string
	top : int
	left : int
	bottom : int
	right : int
	display_name : string
    }



   DeviceStorageInformation  {
	storage_name : string
	total_space : int
	free_space : int
    }



   StackTraceElem  {
	is_native_method : boolean
	method_name : string
	file_name : string
	line_number : int
	class_name : string
    }



   ThreadInformation  {
	thread_id : int
	is_interrupted : boolean
	thread_name : string
	is_alive : boolean
	thread_traces : list<StackTraceElem>
	thread_state : string
	is_daemon : boolean
	priority : int
    }



   DeviceState  {
	other_threads : list<ThreadInformation>
	current_thread : ThreadInformation
	info_map : map<string, string>
	is_connected_to_internet : boolean
	time_zone : string
	storage_information : list<DeviceStorageInformation>
    }



   LogEntry  {
	device_state : DeviceState
	time_fired : int
	values : map<string, string>
	tag : string
	stack_trace : string
	type : string
	message : string
    }



   DatabaseLogEntry  {
	notes : list<string>
	log : list<LogEntry>
	created_at : int
	id : string
	state : string
	device : DeviceInformation
	app_information : AppInformation
	user_information : UserInformation
    }
